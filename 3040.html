1. while ...
$i=0;
while ($i < 10){
    print $i;

}

2. do... while
$i =0;
do{
    print $i;
    $i ++;
} while ($i < 11);

3. for
for ($i=0; $i<10; $i++){
    print $i;
}

4. foreach
$array = array(0,1,2,3);
foreach ($array as $item){
    print $item;
}
